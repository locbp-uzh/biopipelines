# Phase 2 — NeuralPLexer runtime deps.

# torch is re-pinned here even though phase 1 installed it: a transitive dep in
# this phase (pytorch-lightning / fair-esm / e3nn) otherwise pulls the newest
# PyPI torch, which (a) replaces the +cu121 build the PyG wheels link against
# and (b) drags in a CUDA 13 nvidia-*-cu13 wheel stack — breaking both PyG and
# openfold's extension. Pin it to the phase-1 build so nothing upgrades it.
--extra-index-url https://download.pytorch.org/whl/cu121
--find-links https://data.pyg.org/whl/torch-2.1.0+cu121.html
torch==2.1.0

# torch 2.1.0 is built against NumPy 1.x; NumPy 2 triggers `Failed to initialize
# NumPy: _ARRAY_API not found` and breaks tensor<->ndarray interop. The conda
# layer (rdkit) pulls numpy 2, so pin it back here.
numpy<2
pandas

# e3nn: torch is 2.1.0 here, so the modern e3nn (which needs torch>=2.1) is fine
